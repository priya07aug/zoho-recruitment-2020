#include <iostream>

using namespace std;

int
main ()
{
  int size, space, num, i, rnum;
  num = 1;
  cin >> size;
  space = size-1;
  while (!space < 0)
    {
      for (i = 0; i < space; i++)
	{
	  std::cout << " ";
	}
      for (i = 0; i <= size; i++)
	{
	  cout << num;
	  num++;
	}
      space--;
      for (i = 0; i < space; i++)
	{
	  std::cout << " ";
	}
      for (i = 0; i < 5; i++)
	{
	  rnum = num + 5;
	  cout << rnum;
	  rnum--;
	}
      num += 5;
      space--;
    }
    return 0;
}
